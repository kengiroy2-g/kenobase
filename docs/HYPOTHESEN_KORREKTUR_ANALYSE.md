def is_valid_jackpot_ticket(combo: list[int]) -> bool:
    """
    Basierend auf INV-01 bis INV-06.
    ACHTUNG: Hypothetisch, nicht validiert!
    """
    # INV-01: Ziffernprodukt mod 9 = 0
    prod = 1
    for z in combo:
        for d in str(z):
            if d != '0':
                prod *= int(d)
    if prod % 9 != 0:
        return False

    # INV-02: Genau 1 einstellige Zahl
    einstellig = [z for z in combo if z <= 9]
    if len(einstellig) != 1:
        return False

    # INV-03: 6 von 7 Dekaden
    dekaden = set(z // 10 for z in combo)
    if len(dekaden) != 6:
        return False

    # INV-05: Alle Drittel
    if not (any(1 <= z <= 23 for z in combo) and
            any(24 <= z <= 46 for z in combo) and
            any(47 <= z <= 70 for z in combo)):
        return False

    return True
